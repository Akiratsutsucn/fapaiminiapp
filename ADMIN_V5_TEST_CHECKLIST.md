```bash

# 1. AI助手 - 创建会话

  https://api.yourdomain.com/api/admin/ai/sessions

# 2. 爬虫详情
curl -H "Authorization: Bearer YOUR_TOKEN" \
  https://api.yourdomain.com/api/admin/crawler/tasks/123/details

# 3. 审核历史
curl -H "Authorization: Bearer YOUR_TOKEN" \
  "https://api.yourdomain.com/api/admin/data-audit/executions?page=1"

# 4. 数据看板
curl -H "Authorization: Bearer YOUR_TOKEN" \
  "https://api.yourdomain.com/api/admin/dashboard?city_id=0"
```
